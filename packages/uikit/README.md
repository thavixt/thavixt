## Installation

```shell
npm install @thavixt/uikit
```

> Important: This library depends on `tailwind`'s default classes.

## Usage

```tsx
import {Button} from '@thavixt/uikit';

export function ExampleComponent() {
  return (
    <div>
      <Button onClick={() => console.log('you clicked the button')}>
        Hello world!
      </Button>
    </div>
  )
}
```
